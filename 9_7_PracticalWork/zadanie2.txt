#include <iostream>

int main () {
    int pointCount = 0;
    
    std::cout << "\nEnter a long real number: ";
    std::string digit; std::cin >> digit;

    bool logic_digitIn = true;

    if (!((digit[0] >= '0' && digit[0] <= '9') || digit[0] == '-' || digit[0] == '.')) {
        logic_digitIn = false;
    }

    for (int i = 1; i < digit.length(); ++i) {

        if (!(digit[i] >= '0' && digit[i] <= '9') && !(digit[i] == '.')) {
            logic_digitIn = false;
        }

        if (digit[i] == '.') {
            pointCount++;
            if (pointCount > 1) {
                logic_digitIn = false;
            }
        }
    }

    if (logic_digitIn == true) {
        std::cout << "\nYes, your number is real.\n";
    } else {
        std::cout << "\nNO, your number is NOT real.\n";
    }
}
